    myFile="/var /log/httpd/access.log"  
      
    #这里的-x 参数判断$myPath是否存在并且是否具有可执行权限  
    if [ ! -x "$myPath"]; then  
    mkdir "$myPath"  
    fi  
      
    #这里的-d 参数判断$myPath是否存在

    mkdir "$myPath"  
    fi  
      
    #这里的-f参数判断$myFile是否存在  
    if [ ! -f "$myFile" ]; then  
    touch "$myFile"  
    fi
